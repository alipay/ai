# 统一收单交易退款接口 - Java 示例

## 请求示例

```java
package com.java.sdk.demo;

import com.alipay.api.AlipayApiException;
import com.alipay.api.AlipayClient;
import com.alipay.api.DefaultAlipayClient;
import com.alipay.api.AlipayConfig;
import com.alipay.api.domain.AlipayTradeRefundModel;
import com.alipay.api.request.AlipayTradeRefundRequest;
import com.alipay.api.response.AlipayTradeRefundResponse;

public class AlipayTradeRefund {

    public static void main(String[] args) throws AlipayApiException {
        // 初始化SDK（Java 使用 PKCS#8 格式私钥，C#/Python 使用 PKCS#1 格式，请勿混用）
        AlipayClient alipayClient = new DefaultAlipayClient(getAlipayConfig());

        AlipayTradeRefundRequest request = new AlipayTradeRefundRequest();
        AlipayTradeRefundModel model = new AlipayTradeRefundModel();
        model.setOutTradeNo("MERCHANT_ORDER_NO_PLACEHOLDER");  // 商户订单号，与 tradeNo 二选一
        model.setRefundAmount("200.12");            // 退款金额，必填，单位为元
        model.setRefundReason("正常退款");           // 退款原因，选填
        model.setOutRequestNo("OUT_REQUEST_NO_PLACEHOLDER");         // 退款请求号，同一笔交易多次部分退款时必填且必须不同
        request.setBizModel(model);

        AlipayTradeRefundResponse response = alipayClient.execute(request);

        if (response.isSuccess()) {
            // fund_change: Y=发生了资金变化（退款成功），N=未发生资金变化
            System.out.println("资金变化: " + response.getFundChange() + ", 退款金额: " + response.getRefundFee());
        } else {
            System.out.println("调用失败: " + response.getSubCode() + " - " + response.getSubMsg());
        }
    }

    private static AlipayConfig getAlipayConfig() {
        String privateKey  = "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->";
        String alipayPublicKey = "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->";
        AlipayConfig alipayConfig = new AlipayConfig();
        alipayConfig.setServerUrl("https://openapi.alipay.com/gateway.do");
        // 沙箱环境: alipayConfig.setServerUrl("https://openapi-sandbox.dl.alipaydev.com/gateway.do");
        alipayConfig.setAppId("<-- 请填写您的AppId，例如：20190xxxxxx -->");
        alipayConfig.setPrivateKey(privateKey);
        alipayConfig.setFormat("json");
        alipayConfig.setAlipayPublicKey(alipayPublicKey);
        alipayConfig.setCharset("UTF-8");
        alipayConfig.setSignType("RSA2");
        return alipayConfig;
    }
}
```
