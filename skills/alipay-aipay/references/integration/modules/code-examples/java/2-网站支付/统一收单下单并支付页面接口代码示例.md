# 统一收单下单并支付页面接口 - Java 示例

## 请求示例

```java
package com.java.sdk.demo;

import com.alipay.api.AlipayApiException;
import com.alipay.api.AlipayClient;
import com.alipay.api.DefaultAlipayClient;
import com.alipay.api.AlipayConfig;
import com.alipay.api.domain.AlipayTradePagePayModel;
import com.alipay.api.response.AlipayTradePagePayResponse;
import com.alipay.api.request.AlipayTradePagePayRequest;

public class AlipayTradePagePay {

    public static void main(String[] args) throws AlipayApiException {
        // 初始化SDK（Java 使用 PKCS#8 格式私钥，C#/Python 使用 PKCS#1 格式，请勿混用）
        AlipayClient alipayClient = new DefaultAlipayClient(getAlipayConfig());

        AlipayTradePagePayRequest request = new AlipayTradePagePayRequest();
        request.setNotifyUrl("https://your-domain.com/notify");  // 异步通知地址，必须传入
        request.setReturnUrl("https://your-domain.com/return");  // 同步跳转地址，必须传入

        AlipayTradePagePayModel model = new AlipayTradePagePayModel();
        model.setOutTradeNo("MERCHANT_ORDER_NO_PLACEHOLDER");             // 商户订单号，必填
        model.setTotalAmount("88.88");                        // 订单总金额，必填，单位为元
        model.setSubject("Iphone6 16G");                      // 订单标题，必填
        model.setProductCode("FAST_INSTANT_TRADE_PAY");       // 网站支付固定值，必填
        request.setBizModel(model);

        // ⚠️ 网站支付必须使用 pageExecute()，不能使用 execute()
        // ⚠️ 必须使用 POST 模式：alipayClient.pageExecute(request, "POST")
        AlipayTradePagePayResponse response = alipayClient.pageExecute(request, "POST");
        String formHtml = response.getBody();
        // pageExecute 返回的是 HTML 表单字符串，前端需要渲染并自动提交
        System.out.println(formHtml);
    }

    private static AlipayConfig getAlipayConfig() {
        String privateKey  = "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->";
        String alipayPublicKey = "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->";
        AlipayConfig alipayConfig = new AlipayConfig();
        alipayConfig.setServerUrl("https://openapi.alipay.com/gateway.do");
        // 沙箱环境: alipayConfig.setServerUrl("https://openapi-sandbox.dl.alipaydev.com/gateway.do");
        alipayConfig.setAppId("<-- 请填写您的AppId，例如：20190xxxxxx -->");
        alipayConfig.setPrivateKey(privateKey);
        alipayConfig.setFormat("json");
        alipayConfig.setAlipayPublicKey(alipayPublicKey);
        alipayConfig.setCharset("UTF-8");
        alipayConfig.setSignType("RSA2");
        return alipayConfig;
    }
}
```
