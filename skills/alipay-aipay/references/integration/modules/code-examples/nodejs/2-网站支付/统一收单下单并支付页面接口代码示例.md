# 统一收单下单并支付页面接口 - Node.js 示例

## 请求示例

```javascript
const { AlipaySdk } = require("alipay-sdk");

const alipaySdk = new AlipaySdk({
  appId: "<-- 请填写您的AppId，例如：20190xxxxxx -->",
  privateKey: "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->", // Node.js 使用 PKCS#1 格式私钥
  alipayPublicKey: "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->",
  gateway: "https://openapi.alipay.com/gateway.do",
  signType: "RSA2",
  // 第三方代调用模式下请设置 appAuthToken
  // appAuthToken: "<-- 请填写应用授权令牌 -->",
});

// ⚠️ 网站支付必须使用 pageExec() 方法，不能使用 exec()
// ⚠️ pageExec() 第二个参数必须传 'POST'
// ⚠️ notifyUrl 在真实生产或公网联调时传入；本地生产参数验收模式可临时不传，但仍需实现通知处理代码
// ⚠️ 本地没有公网 HTTPS notifyUrl 时，直接省略 notifyUrl，不要向 SDK 传 undefined、空字符串或占位 URL
// ⚠️ returnUrl 默认传入，仅在用户明确关闭同步回跳时省略
const notifyUrl = (process.env.ALIPAY_NOTIFY_URL || "").trim();
const requestOptions = {
  returnUrl: "https://your-domain.com/return", // 默认传入并替换为已验证结果页；仅用户明确关闭同步回跳时省略
  bizContent: {
    out_trade_no: "MERCHANT_ORDER_NO_PLACEHOLDER",         // 商户订单号，必填
    total_amount: "88.88",                      // 订单总金额，必填，单位为元，精确到小数点后两位
    subject: "Iphone6 16G",                     // 订单标题，必填
    product_code: "FAST_INSTANT_TRADE_PAY",     // 网站支付固定值，必填
  },
};
if (notifyUrl) {
  requestOptions.notifyUrl = notifyUrl;
}

const result = alipaySdk.pageExec(
  "alipay.trade.page.pay",
  "POST",
  requestOptions
);

// pageExec 返回的是 HTML 表单字符串，前端需要渲染并自动提交
// ⚠️ 禁止用 URL 直接跳转，否则页面只会显示参数而无法跳转支付宝
console.log(result);
```

### 前端渲染支付表单

```javascript
// pageExec 返回的 result 是包含自动提交脚本的 HTML <form> 字符串
// 前端必须将此 HTML 渲染到页面并自动提交表单

// ✅ 正确写法（以浏览器端 JavaScript 为例）
if (data.paymentHtml) {
  const container = document.createElement('div');
  container.innerHTML = data.paymentHtml;
  container.style.position = 'fixed';
  container.style.top = '0';
  container.style.left = '0';
  container.style.width = '100%';
  container.style.height = '100%';
  container.style.zIndex = '9999';
  container.style.background = 'white';
  document.body.appendChild(container);
  const form = container.querySelector('form');
  if (form) form.submit();
}

// ❌ 错误写法 — 禁止用 URL 直接跳转
// window.location.href = data.paymentHtml; // 页面只会显示参数，无法跳转支付宝
```
