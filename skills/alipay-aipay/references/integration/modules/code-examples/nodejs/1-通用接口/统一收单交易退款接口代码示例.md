# 统一收单交易退款接口 - Node.js 示例

## 请求示例

```javascript
const { AlipaySdk } = require("alipay-sdk");

const alipaySdk = new AlipaySdk({
  appId: "<-- 请填写您的AppId，例如：20190xxxxxx -->",
  privateKey: "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->", // Node.js 使用 PKCS#1 格式私钥
  alipayPublicKey: "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->",
  gateway: "https://openapi.alipay.com/gateway.do",
  signType: "RSA2",
  // 第三方代调用模式下请设置 appAuthToken
  // appAuthToken: "<-- 请填写应用授权令牌 -->",
});

const result = await alipaySdk.exec("alipay.trade.refund", {
  bizContent: {
    out_trade_no: "MERCHANT_ORDER_NO_PLACEHOLDER",  // 商户订单号，与 trade_no 二选一
    refund_amount: "200.12",             // 退款金额，必填，单位为元
    refund_reason: "正常退款",            // 退款原因，选填
    out_request_no: "OUT_REQUEST_NO_PLACEHOLDER",         // 退款请求号，同一笔交易多次部分退款时必填且必须不同
  },
});

if (result.code === "10000") {
  console.log("调用成功");
} else {
  console.log("调用失败");
}
```
