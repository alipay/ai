# 统一收单交易关闭接口 - Node.js 示例

## 请求示例

```javascript
const { AlipaySdk } = require("alipay-sdk");

const alipaySdk = new AlipaySdk({
  appId: "<-- 请填写您的AppId，例如：20190xxxxxx -->",
  privateKey: "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->", // Node.js 使用 PKCS#1 格式私钥
  alipayPublicKey: "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->",
  gateway: "https://openapi.alipay.com/gateway.do",
  signType: "RSA2",
  // 第三方代调用模式下请设置 appAuthToken
  // appAuthToken: "<-- 请填写应用授权令牌 -->",
});

// 交易关闭用于关闭未支付的订单
// 注意：交易关闭与交易撤销是不同接口 —— 关闭用于未支付订单，撤销用于已支付订单（仅限支付当天）
const result = await alipaySdk.exec("alipay.trade.close", {
  bizContent: {
    out_trade_no: "MERCHANT_ORDER_NO_PLACEHOLDER",  // 商户订单号，与 trade_no 二选一
  },
});

if (result.code === "10000") {
  console.log("调用成功");
} else {
  console.log("调用失败");
}
```

## 响应示例
### 正常示例
```json
{
    "code": "10000",
    "msg": "Success",
    "trade_no": "ALIPAY_TRADE_NO_PLACEHOLDER",
    "out_trade_no": "MERCHANT_ORDER_NO_PLACEHOLDER"
}
```
### 异常示例
```json
{
    "code": "20000",
    "msg": "Service Currently Unavailable",
    "sub_code": "isp.unknow-error",
    "sub_msg": "系统繁忙"
}
```
