# 统一收单交易退款查询接口 - PHP 示例

## 请求示例

```php
<?php
require_once __DIR__ . '/aop/AopClient.php';
require_once __DIR__ . '/aop/request/AlipayTradeFastpayRefundQueryRequest.php';

// 初始化 SDK（PHP 使用 PKCS#1 格式私钥，Java 使用 PKCS#8 格式，请勿混用）
$aop = new AopClient();
$aop->gatewayUrl = 'https://openapi.alipay.com/gateway.do';
// 沙箱环境: $aop->gatewayUrl = 'https://openapi-sandbox.dl.alipaydev.com/gateway.do';
$aop->appId = '<-- 请填写您的AppId，例如：20190xxxxxx -->';
$aop->rsaPrivateKey = '<-- 请填写您的应用私钥（PKCS#1格式），例如：MIIE...（脱敏示例） -->';
$aop->alipayrsaPublicKey = '<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->';
$aop->format = 'json';
$aop->postCharset = 'UTF-8';
$aop->signType = 'RSA2';

// 构造请求参数
$request = new AlipayTradeFastpayRefundQueryRequest();
$bizContent = [
    'out_trade_no' => 'MERCHANT_ORDER_NO_PLACEHOLDER',     // 商户订单号
    'out_request_no' => 'OUT_REQUEST_NO_PLACEHOLDER',          // 退款请求号，必填
];
$request->setBizContent(json_encode($bizContent, JSON_UNESCAPED_UNICODE));

// 调用接口
$response = $aop->execute($request);
$responseApiName = str_replace('.', '_', $request->getApiMethodName()) . '_response';
$result = $response->$responseApiName;

if (!empty($result->code) && $result->code == '10000') {
    // refund_status: REFUND_SUCCESS 表示退款成功
    echo "退款状态: " . $result->refund_status . ", 退款金额: " . $result->refund_amount;
} else {
    echo "调用失败: " . $result->sub_code . " - " . $result->sub_msg;
}
```
