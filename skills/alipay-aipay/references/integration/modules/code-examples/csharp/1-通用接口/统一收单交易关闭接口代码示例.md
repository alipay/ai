# 统一收单交易关闭接口 - C# 示例

## 请求示例

```csharp
using Aop.Api;
using Aop.Api.Request;
using Aop.Api.Response;
using Aop.Api.Domain;

// 初始化SDK（C#/.NET 使用 PKCS#1 格式私钥，Java 使用 PKCS#8 格式，请勿混用）
IAopClient alipayClient = new DefaultAopClient(GetAlipayConfig());

var request = new AlipayTradeCloseRequest();
var model = new AlipayTradeCloseModel
{
    OutTradeNo = "MERCHANT_ORDER_NO_PLACEHOLDER",  // 商户订单号，与 TradeNo 二选一
    // TradeNo = "ALIPAY_TRADE_NO_PLACEHOLDER",  // 支付宝交易号，同时传时以 TradeNo 为准
};
request.SetBizModel(model);

AlipayTradeCloseResponse response = alipayClient.Execute(request);

if (!response.IsError)
{
    Console.WriteLine("交易关闭成功");
}
else
{
    Console.WriteLine($"调用失败: {response.SubCode} - {response.SubMsg}");
}

private static AlipayConfig GetAlipayConfig()
{
    string privateKey  = "<-- 请填写您的应用私钥，例如：MIIE...（脱敏示例） -->";
    string alipayPublicKey = "<-- 请填写您的支付宝公钥，例如：MIIB...（脱敏示例） -->";
    AlipayConfig alipayConfig = new AlipayConfig();
    alipayConfig.ServerUrl = "https://openapi.alipay.com/gateway.do";
    // 沙箱环境: alipayConfig.ServerUrl = "https://openapi-sandbox.dl.alipaydev.com/gateway.do";
    alipayConfig.AppId = "<-- 请填写您的AppId，例如：20190xxxxxx -->";
    alipayConfig.PrivateKey = privateKey;
    alipayConfig.Format = "json";
    alipayConfig.AlipayPublicKey = alipayPublicKey;
    alipayConfig.Charset = "UTF-8";
    alipayConfig.SignType = "RSA2";
    return alipayConfig;
}
```
